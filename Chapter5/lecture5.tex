export default function Home(){
    const [count, setCount] = useState(0)
    return (
        <div className={styles.main}>
            <button onClick={() => setCount(count + 1)}>Increase Count</button>
            {count}
        </div>
    );
}
\end{lstlisting}
\end{frame}

\section{Use Effect}
\frame{\frametitle{Use Effect}
    When we want to have a variable which reacts to the change, we can use useEffect 
}

\begin{frame}[fragile]
\frametitle{Use Effect Counter Example}
\begin{lstlisting}
export default function Home(){
    const [count, setCount] = useState(0)

    useEffect(() => {
        console.log(count);
    }, [count]);
    ...
}
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]
\frametitle{Use Effect Counter Example without variable capture}
\begin{lstlisting}

    const [count, setCount] = useState(0)

    useEffect(() => {
        console.log(count);
    });
    ...
}
\end{lstlisting}
\end{frame}
