\section{Components}
\frame{\frametitle{Components Example}
    Components are the bread and butter of React. The way you build any part of the website is in a component. They have
    their own logic and appearance.
}

\begin{frame}[fragile]
\subsection{My Button Component}
\frametitle{My Button Component}
\begin{lstlisting}
function MyButton() {
  return (
      <button>{"I'm a button"}</button>
  );
}
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]
\subsection{My Button Component Home}
\frametitle{My Button Component Home}
\begin{lstlisting}
export default function Home() {
  return (
  <>
    <MyButton/>
  </>
 );
}\end{lstlisting}
\end{frame}

\section{HTML vs JSX}
\frame{\frametitle{HTML vs JSX}
    HTML tags use lowercase button, whereas JSX tags use uppercase MyButton. For a 
    list of html tags checkout \href{w3schools}w3schools
}
